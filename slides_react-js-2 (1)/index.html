export default class ProductListPage extends React.Component {
  // 불러온 데이터를 state에 저장
  constructor(props) {
    super(props)
    
    this.state = {
       loading: false,
       products: null,
       error: null
    }
  }
  
  componentDidMount() {
    this.setState({
      ...this.state,
      loading: true
    }, () => {
      fetch(API_URL)
        .then(res => res.json())
        .then(products => this.setState({ products }))
        .catch(e => this.setState({ error: e.message }))
        .finally(() => this.setState({ ...this.state, loading: false }))
    })
    
  }
  
  render() {
    if (this.state.loading) {
      return <div>loading...</div>
    }
    
    if (this.state.error) {
      return <div>데이터 로딩에 실패했습니다.</div>
    }
    
    // 서버에서 불러온 데이터 결과가 null이거나 빈 배열일 때
    if (!this.state.products || this.state.products.length === 0) {
      return <div>상품이 없습니다.</div>
    }
    
    // rendering process
    return this.state.products....
  }
}

SPA

Single Page Application

  • 전통적인 웹 페이지처럼 페이지를 이동할 때마다 서버(PHP, JSP, Node.js등)에 요청을 보내 화면을 전부 새로 렌더링 하는 방식이 아닌, 동적으로 변경될 페이지만 다시 그리는 방식
  • 화면 전체를 다시 불러올 필요가 없기 때문에 화면 전환 속도가 빠름
  • 단순 웹 페이지가 아닌 애플리케이션과 같은 사용성을 가짐
  • 화면 새로고침이 일어나지 않기 때문에 연속성이 필요한 화면에서 유용
  • 렌더링 로직을 client에 모두 둘 수 있기 때문에 일관된 렌더링 로직을
    가질 수 있음

Single Page Application

  • 애플리케이션 전체에서 사용될 css, js를 모두 불러와야 하기 때문에
    최초 로딩이 느릴 수 있음
  • 최근에는 각 페이지별로 css, js를 나눠서 동적 로딩하는 방법으로
    최적화
  • 전통적인 방식의 웹 페이지는 화면 이동 시
    새로고침이 강제되기 때문에 메모리 누수가 되어도 여파가 적으나
    SPA는 화면을 이동해도 새로고침 되지 않기 때문에 문제가 될 수 있음
  • 페이지별로 meta og 관련 처리를 하려면 별도의 조치가 필요
    • serverless 함수, Server Side Rendering 등
